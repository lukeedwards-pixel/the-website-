<script>

  const blade = document.getElementById('blade');

  const handle = document.getElementById('handle');

  const colorSelect = document.getElementById('colorSelect');

  const handleSelect = document.getElementById('handleSelect');


  function updateBladeColor(color) {

    blade.style.backgroundColor = color;

    blade.style.boxShadow = `0 0 40px ${color}, 0 0 80px ${color}, 0 0 120px ${color}`;

  }


  function updateHandleStyle(style) {

    // Remove previous handle style classes

    handle.classList.remove('style1', 'style2', 'style3');

    // Add selected handle style class

    handle.classList.add(style);

  }


  colorSelect.addEventListener('change', (e) => {

    updateBladeColor(e.target.value);

  });


  handleSelect.addEventListener('change', (e) => {

    updateHandleStyle(e.target.value);

  });


  // Initialize with default values

  updateBladeColor(colorSelect.value);

  updateHandleStyle(handleSelect.value);

</script>
